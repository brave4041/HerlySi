01 LATIHAN 1: CONNECTION POOLING
Dartpad

Dalam rekayasa perangkat lunak, pembuatan koneksi database adalah proses yang berat dan mahal. Database Connection Pooling menghemat sumber daya dengan menyimpan sekumpulan koneksi yang sudah terbuka (pool) dan menggunakannya kembali (reusing) saat dibutuhkan. Di sini, kita membatasi jumlah koneksi aktif maksimal 3.

class ConnectionFactory {
  final String id;
  final String dbName;
  final String host;
  final int port;
  bool isReleased = true;

  static const int maxConnections = 3;
  static final Map<String, List<ConnectionFactory>> _pool = {};

  ConnectionFactory._(this.id, this.dbName, this.host, this.port);

  factory ConnectionFactory(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}:$host:$port';
    
    if (!_pool.containsKey(key)) {
      _pool[key] = [];
    }

    final poolList = _pool[key]!;

    for (var conn in poolList) {
      if (conn.isReleased) {
        conn.isReleased = false;
        print('[Pool Hit] Menggunakan kembali koneksi idle: ${conn.id} untuk $key');
        return conn;
      }
    }

    if (poolList.length < maxConnections) {
      final connId = 'conn_${dbName.toLowerCase()}_${poolList.length + 1}';
      final newConn = ConnectionFactory._(connId, dbName, host, port);
      newConn.isReleased = false;
      poolList.add(newConn);
      print('[Pool Miss] Membuat koneksi database baru: $connId untuk $key');
      return newConn;
    }

    throw StateError('Batas maksimum koneksi database tercapai ($maxConnections) untuk pool $key.');
  }

  void query(String sql) {
    if (isReleased) {
      throw StateError('Koneksi $id telah dilepas! Hubungkan kembali sebelum query.');
    }
    print('[$id] Menjalankan query: "$sql" pada database "$dbName" ($host:$port)');
  }

  static void release(ConnectionFactory conn) {
    conn.isReleased = true;
    print('[Pool Release] Koneksi ${conn.id} berhasil dilepaskan kembali ke pool.');
  }

  static void printStatus(String dbName, {String host = 'localhost', int port = 5432}) {
    final key = '${dbName.toLowerCase()}:$host:$port';
    final poolList = _pool[key];
    print('=== STATUS POOL KONEKSI ($key) ===');
    if (poolList == null || poolList.isEmpty) {
      print('Belum ada koneksi yang terdaftar.');
    } else {
      for (var conn in poolList) {
        print(' - ${conn.id}: [${conn.isReleased ? "IDLE / TERSEDIA" : "ACTIVE / SEDANG DIGUNAKAN"}]');
      }
    }
    print('=================================');
  }

  @override
  String toString() {
    return 'ConnectionFactory(id: $id, database: $dbName, host: $host, port: $port, status: ${isReleased ? "Idle" : "Active"})';
  }
}

import 'libconnection_factory.dart';

void main() {
  print('=== UJI COBA DATABASE CONNECTION POOLING ===\n');

  final dbName = 'pos_system';

  print('--- Mengambil koneksi pertama ---');
  final conn1 = ConnectionFactory(dbName);
  conn1.query('SELECT * FROM users');
  print('');

  print('--- Mengambil koneksi kedua ---');
  final conn2 = ConnectionFactory(dbName);
  conn2.query('SELECT * FROM products');
  print('');

  ConnectionFactory.printStatus(dbName);
  print('');

  print('--- Melepaskan koneksi pertama ---');
  ConnectionFactory.release(conn1);
  print('');

  ConnectionFactory.printStatus(dbName);
  print('');

  print('--- Mengambil koneksi ketiga (harus menggunakan kembali conn1) ---');
  final conn3 = ConnectionFactory(dbName);
  conn3.query('UPDATE stock SET quantity = 100 WHERE id = 1');
  print('Apakah conn1 identik dengan conn3? ${identical(conn1, conn3)}');
  print('');

  print('--- Mengambil koneksi keempat ---');
  final conn4 = ConnectionFactory(dbName);
  conn4.query('INSERT INTO transaction_history VALUES (1, 50000)');
  print('');

  ConnectionFactory.printStatus(dbName);
  print('');

  print('--- Mencoba mengambil koneksi kelima (Harus memicu error) ---');
  try {
    final conn5 = ConnectionFactory(dbName);
    conn5.query('SELECT * FROM audit_logs');
  } catch (e) {
    print('ERROR DITANGKAP: $e');
  }
}

Menggunakan Factory Constructor untuk menerapkan pola Subclass Selection. Klien hanya berinteraksi dengan kelas induk Notification, sementara factory constructor di dalamnya secara dinamis memutuskan instansi subclass mana yang dibuat (Android, iOS, Web, atau Windows) berdasarkan parameter platform yang dimasukkan.

abstract class NotificationFactory {
  String get message;
  void send();

  factory NotificationFactory(String platform, String message) {
    switch (platform.toLowerCase()) {
      case 'android':
        return AndroidNotification(message);
      case 'ios':
        return IOSNotification(message);
      case 'web':
        return WebNotification(message);
      case 'windows':
        return WindowsNotification(message);
      default:
        throw ArgumentError('Platform tidak didukung: "$platform". Silakan pilih android, ios, web, atau windows.');
    }
  }
}

class AndroidNotification implements NotificationFactory {
  @override
  final String message;

  AndroidNotification(this.message);

  @override
  void send() {
    print('[Android FCM] Mengirim Firebase Cloud Messaging push notification:');
    print('  -> Message: "$message"');
  }
}

class IOSNotification implements NotificationFactory {

03 LATIHAN 3: SHAPE CACHING
Dartpad

Mengimplementasikan pola desain Flyweight Pattern menggunakan caching statis di dalam Factory Constructor. Jika klien meminta instansi Shape (lingkaran, persegi, segitiga) dengan properti yang identik (tipe, ukuran, dan warna sama), factory akan mengembalikan objek yang sama dari cache statis, menghemat penggunaan memori secara drastis.

FILE: libshape_factory.dart
// Nama   : Herly Afrizal
// Nim    : 251410002
// Kelas  : SI2A

abstract class ShapeFactory {
  String get type;
  double get size;
  String get color;
  void draw();

  static final Map<String, ShapeFactory> _cache = {};

  factory ShapeFactory(String type, double size, String color) {
    if (size <= 0) {
      throw ArgumentError('Ukuran bentuk (size) harus lebih besar dari 0!');
    }
    if (color.trim().isEmpty) {
      throw ArgumentError('Warna bentuk tidak boleh kosong!');
    }

    final key = '${type.toLowerCase()}:${size}:${color.trim().toLowerCase()}';

    if (_cache.containsKey(key)) {
      print('[Cache Hit] Menggunakan kembali instansi Shape yang ada: $key');
      return _cache[key]!;
    }

    print('[Cache Miss] Membuat instansi Shape baru untuk: $key');
    ShapeFactory newShape;
    switch (type.toLowerCase()) {
      case 'circle':
        newShape = Circle._(size, color);
        break;
      case 'square':
        newShape = Square._(size, color);
        break;
      case 'triangle':
        newShape = Triangle._(size, color);
        break;
      default:
        throw ArgumentError('Tipe bentuk "$type" tidak didukung! Pilih circle, square, atau triangle.');
    }

    _cache[key] = newShape;
    return newShape;
  }

  static int get cacheSize => _cache.length;

  static void clearCache() {
    _cache.clear();
    print('[Cache Clear] Cache bentuk berhasil dibersihkan.');
  }
}

class Circle implements ShapeFactory {
  @override
  final String type = 'Circle';
  @override
  final double size;
  @override
  final String color;

  Circle._(this.size, this.color);

  @override
  void draw() {
    print('Menggambar Lingkaran ($color) dengan jari-jari: $size');
  }
}

class Square implements ShapeFactory {
  @override
  final String type = 'Square';
  @override
  final double size;
  @override
  final String color;

  Square._(this.size, this.color);

  @override
  void draw() {
    print('Menggambar Persegi ($color) dengan panjang sisi: $size');
  }
}

class Triangle implements ShapeFactory {
  @override
  final String type = 'Triangle';
  @override
  final double size;
  @override
  final String color;

  Triangle._(this.size, this.color);

  @override
  void draw() {
    print('Menggambar Segitiga ($color) dengan alas/tinggi: $size');
  }
}

04 CHALLENGE: ANIMAL FACTORY
Challenge

Mengintegrasikan seluruh konsep dalam satu rancangan AnimalFactory yang kompleks. Fitur-fitur yang diterapkan mencakup:

  • Subclass Selection: Memilih secara dinamis subclass Dog, Cat, atau Bird berdasarkan tipe input.
  • Strict Parameter Validation: Memastikan nama tidak boleh kosong (menggunakan trim()) dan umur tidak boleh kurang dari 0.
  • Flyweight Multi-Property Caching: Menghasilkan cache dinamis dengan key terenkripsi properti unik: tipe_hewan + nama + umur. Hewan identik akan didapat kembali secara instan tanpa pemborosan memori.
FILE: libanimal_factory.dart

// Nim    : 251410002
// Kelas  : SI2A

abstract class AnimalFactory {
  String get name;
  int get age;
  String get sound;
  void speak();

  static final Map<String, AnimalFactory> _cache = {};

  factory AnimalFactory(String type, String name, int age) {
    if (name.trim().isEmpty) {
      throw ArgumentError('Nama hewan tidak boleh kosong!');
    }
    if (age < 0) {
      throw ArgumentError('Umur hewan tidak boleh kurang dari 0!');
    }

    final key = '${type.toLowerCase()}:${name.trim().toLowerCase()}:$age';

    if (_cache.containsKey(key)) {
      print('[Cache Hit] Mengambil instansi Animal dari cache: $key');
      return _cache[key]!;
    }

    print('[Cache Miss] Membuat instansi Animal baru untuk: $key');
    AnimalFactory newAnimal;
    switch (type.toLowerCase()) {
      case 'dog':
        newAnimal = Dog._(name, age);
        break;
      case 'cat':
        newAnimal = Cat._(name, age);
        break;
      case 'bird':
        newAnimal = Bird._(name, age);
        break;
      default:
        throw ArgumentError('Tipe hewan "$type" tidak didukung! Pilih dog, cat, atau bird.');
    }

    _cache[key] = newAnimal;
    return newAnimal;
  }

  static int get cacheSize => _cache.length;

  static void clearCache() {
    _cache.clear();
    print('[Cache Clear] Cache hewan berhasil dibersihkan.');
  }
}

class Dog implements AnimalFactory {
  @override
  final String name;
  @override
  final int age;
  @override
  final String sound = 'Guk Guk!';

  Dog._(this.name, this.age);

  @override
  void speak() {
    print('[$name - Anjing ($age tahun)]: $sound');
  }
}

class Cat implements AnimalFactory {
