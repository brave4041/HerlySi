01 LATIHAN: CIRCLE REDIRECTING CONSTRUCTORS
Dartpad

Mengimplementasikan Redirecting Constructors pada kelas Circle. Konstruktor utama menerima parameter radius yang dilengkapi validasi parameter, sedangkan tiga konstruktor pengarah (fromDiameter, fromKeliling, dan fromLuas) mengalihkan instansiasi ke konstruktor utama tanpa memiliki body terpisah (menerapkan prinsip DRY - Don't Repeat Yourself).

FILE: libcircle.dart
// Nama   : Herly Afrizal

import 'dart:math' as math;

class Circle {
  final double radius;

  Circle(this.radius) {
    if (radius <= 0) {
      throw ArgumentError('Radius harus lebih besar dari 0!');
    }
  }

  Circle.fromDiameter(double diameter) : this(diameter / 2);

  Circle.fromKeliling(double keliling) : this(keliling / (2 * math.pi));

  Circle.fromLuas(double luas) : this(math.sqrt(luas / math.pi));

  double get diameter => radius * 2;
  double get keliling => 2 * math.pi * radius;
  double get luas => math.pi * radius * radius;

  @override
  String toString() {
    return 'Circle(radius: ${radius.toStringAsFixed(2)}, '
        'diameter: ${diameter.toStringAsFixed(2)}, '
        'keliling: ${keliling.toStringAsFixed(2)}, '
        'luas: ${luas.toStringAsFixed(2)})';
  }
}

import 'libcircle.dart';

void main() {
  print('=== UJI COBA CIRCLE CLASS DENGAN REDIRECTING CONSTRUCTORS ===\n');

  print('--- 1. Membuat Circle 1 melalui Radius (Main Constructor: 5.0) ---');
  final c1 = Circle(5.0);
  print(c1);
  print('');

  print('--- 2. Membuat Circle 2 melalui Diameter (Redirecting: 10.0) ---');
  final c2 = Circle.fromDiameter(10.0);
  print(c2);
  print('Apakah radius c2 sama dengan c1? ${c1.radius == c2.radius}');
  print('');

  print('--- 3. Membuat Circle 3 melalui Keliling (Redirecting: 31.41592653589793) ---');
  final c3 = Circle.fromKeliling(31.41592653589793);
  print(c3);
  print('Apakah radius c3 mendekati c1? ${(c1.radius - c3.radius).abs() < 1e-9}');
  print('');

  print('--- 4. Membuat Circle 4 melalui Luas (Redirecting: 78.53981633974483) ---');
  final c4 = Circle.fromLuas(78.53981633974483);
  print(c4);
  print('Apakah radius c4 mendekati c1? ${(c1.radius - c4.radius).abs() < 1e-9}');
  print('');

  print('--- 5. Menguji Validasi Radius Negatif ---');
  try {
    final invalidCircle = Circle(-2.0);
    print(invalidCircle);
  } catch (e) {
    print('ERROR DITANGKAP: $e');
  }
  print('');

  print('--- 6. Menguji Validasi Diameter Nol ---');
  try {
    final invalidDiameter = Circle.fromDiameter(0);
    print(invalidDiameter);
  } catch (e) {
    print('ERROR DITANGKAP: $e');
  }
  print('');

  print('--- 7. Menguji Validasi Keliling Negatif ---');
  try {
    final invalidKeliling = Circle.fromKeliling(-10.0);
    print(invalidKeliling);
  } catch (e) {
    print('ERROR DITANGKAP: $e');
  }
  print('');

  print('--- 8. Menguji Validasi Luas Negatif ---');
  try {
    final invalidLuas = Circle.fromLuas(-50.0);
    print(invalidLuas);
  } catch (e) {
    print('ERROR DITANGKAP: $e');
  }
}
