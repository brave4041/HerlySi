01 RESUME: KONSTRUKTOR & ENKAPSULASI

Materi ini membahas dua pilar penting dalam pemrograman berorientasi objek menggunakan bahasa Dart: bagaimana objek diinisialisasi melalui konstruktor dan bagaimana data dilindungi melalui enkapsulasi.

02 KONSTRUKTOR (CONSTRUCTOR)

Konstruktor adalah method khusus yang dipanggil saat objek dibuat. Dart menyediakan beberapa jenis konstruktor:

  • Default Constructor: Dibuat otomatis jika tidak didefinisikan secara manual.
  • Parameterized Constructor: Menerima parameter untuk menginisialisasi field secara langsung.
  • Named Constructor: Memberikan nama tambahan untuk cara alternatif pembuatan objek.
  • Constant Constructor: Membuat objek immutable untuk efisiensi memori.
  • Initializer List: Digunakan untuk validasi field sebelum body konstruktor dijalankan.

03 ENKAPSULASI (ENCAPSULATION)

Konsep menyembunyikan detail implementasi untuk keamanan dan kontrol data.

  • Private Field: Menggunakan prefix underscore (_) untuk membatasi akses hanya dalam satu file.
  • Kontrol Akses: Memastikan integritas data melalui mekanisme internal.
  • Fleksibilitas: Mempermudah perubahan logika internal tanpa merusak kode luar.

04 GETTER DAN SETTER

Mekanisme untuk mengakses dan memodifikasi data private dengan kontrol tambahan.

  • Getter: Mengambil nilai data private (Contoh: get nama => _name;).
  • Setter: Mengubah nilai data private dengan validasi (Contoh: set nama(val) { ... }).

class Mahasiswa {
  String? nama;
  int? umur;

  // Parameterized constructor
  Mahasiswa(this.nama, this.umur);

  // Named constructor dari Map
  Mahasiswa.dariMap(Map<String, dynamic> map) {
    nama = map['nama'];
    umur = map['umur'];
  }

  // Named constructor tanpa nama
  Mahasiswa.tanpaNama() {
    nama = 'Anonymous';
    umur = 0;
  }

  @override
  String toString() => 'Mahasiswa{nama: $nama, umur: $umur}';
}

void main() {
  print('--- Praktik 1: Kelas Mahasiswa ---');
  var m1 = Mahasiswa('Herly', 20);
  var m2 = Mahasiswa.dariMap({'nama': 'Afrizal', 'umur': 22});
  var m3 = Mahasiswa.tanpaNama();

  print(m1);
  print(m2);
  print(m3);
}

class User {
  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email) {
    if (_username.length < 5) _username = 'invalid';
    if (_password.length < 8) _password = 'invalid';
    if (!_email.contains('@')) _email = 'invalid';
  }

  String get username => _username;
  String get email => _email;

  set username(String value) {
    if (value.length >= 5) _username = value;
    else print('Username harus minimal 5 karakter.');
  }

  set password(String value) {
    if (value.length >= 8) _password = value;
    else print('Password harus minimal 8 karakter.');
  }

  set email(String value) {
    if (value.contains('@')) _email = value;
    else print('Email harus mengandung @.');
  }

  void info() {
    print('Username: $_username, Email: $_email');
  }
}

void main() {
  print('\n--- Praktik 2: Kelas User dengan Validasi ---');
  var user = User('herly', 'rahasia123', 'herly@mail.com');
  user.info();

  print('\nMengubah username menjadi "her" (gagal)');
  user.username = 'her';
  
  print('\nMengubah username menjadi "herlyafrizal" (berhasil)');
  user.username = 'herlyafrizal';
  user.info();

  print('\nMengubah password menjadi "short" (gagal)');
  user.password = 'short';

  print('\nMengubah email menjadi "herly.mail.com" (tanpa @ - gagal)');
  user.email = 'herly.mail.com';

  print('\nMengubah email menjadi "herly@mail.com" (berhasil)');
  user.email = 'herly@mail.com';
  user.info();
}

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo) {
    if (_saldo < 0) _saldo = 0;
  }

  String get nomorRekening => _nomorRekening;
  double get saldo => _saldo;

  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print('Deposit Rp $jumlah berhasil. Saldo: Rp $_saldo');
    }
  }

  void tarik(double jumlah) {
    if (jumlah > 0 && _saldo >= jumlah) {
      _saldo -= jumlah;
      print('Tarik Rp $jumlah berhasil. Saldo: Rp $_saldo');
    } else {
      print('Transaksi gagal: Saldo tidak mencukupi atau jumlah tidak valid.');
    }
  }

  void cekSaldo() {
    print('Saldo rekening $_nomorRekening: Rp $_saldo');
  }
}

void main() {
  print('\n--- Latihan: RekeningBank ---');
  var rek = RekeningBank('123-456-789', 500000);
  rek.cekSaldo();
  rek.deposit(200000);
  rek.tarik(100000);
  rek.cekSaldo();
}
